283. Move Zeroes

class Solution {
    public void moveZeroes(int[] nums) {
        //[0...k) store non-zero
        int k = 0;
        for(int i = 0 ; i< nums.length ; i++){
            if(nums[i] != 0){
                nums[k] = nums[i];
                k++;
            }
        }
        
        for(int j = k; j < nums.length;j++){
            nums[j] = 0;
        }

    }
}

26. Remove Duplicates from Sorted Array

sorted array

class Solution {
    public int removeDuplicates(int[] nums) {
        if (nums.length == 0) return 0;
        // nums[0...k]store non-duplicate
        int k = 0;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[k]) {
                k++;
                nums[k] = nums[i];
            }
        }
        return k + 1;
    }
}

215. Kth Largest Element in an Array

  • 快排-速度快
  • 堆排序-支持数据的动态增长
import java.util.Random;

class Solution {
    //1 ≤ k ≤ 数组的长度
    public int findKthLargest(int[] nums, int k) {
        //找第k大转换为找有序排列时的nums[k]
        k = nums.length - k;
        Random rdm = new Random();
        int num = nums[findKthLargestR(nums, 0, nums.length - 1, k, rdm)];
        return num;
    }


    //arr[l...r]中寻找nums[k] 并返回nums[k]的位置
    private int findKthLargestR(int[] nums, int l, int r, int k, Random rdm) {
        if (l > r) {
            return -1;
        }

        int p = partition(nums, l, r, rdm);

       if (p < k) {
           return findKthLargestR(nums, p + 1, r, k,rdm);
        } else if (p > k) {
           return findKthLargestR(nums, l, p - 1,  k,rdm);
        }

        return p;
    }

    //arr[l...r]中寻找nums[k] 并返回nums[k]的位置 非递归
    private int findKthLargest(int[] nums, int l, int r, int k, Random rdm) {
        if (l > r) {
            return -1;
        }

        int p = partition(nums, l, r, rdm);

        while (p != k) {
            if (p < k) {
                p = partition(nums, p + 1, r, rdm);
            } else if (p > k) {
                p = partition(nums, l, p - 1, rdm);
            }
        }
        return p;
    }

    private int partition(int[] arr, int l, int r, Random rdm) {
        //随机索引
        int p = rdm.nextInt(r - l + 1) + l;
        //保持arr[l]为pivot
        swap(arr, l, p);

        // arr[l+1...i-1]<v , arr[j+1...r]>v, i is working-pointer
        int i = l + 1;
        int j = r;
        while (true) {
            while (i <= j && arr[i] - arr[l] < 0) {
                i++;
            }

            while (i <= j && arr[j] - arr[l] > 0) {
                j--;
            }

            if (i >= j) {
                break;
            }
            swap(arr, i, j);
            i++;
            j--;

        }
        swap(arr, l, j);
        return j;
    }

    private static void swap(int[] arr, int i, int minIndex) {
        int tmp = arr[i];
        arr[i] = arr[minIndex];
        arr[minIndex] = tmp;
    }

    public static void main(String[] args) {

        int[] nums = {3};
        System.out.println(new Solution().findKthLargest(nums, 1));
    }

}

167. Two Sum II - Input array is sorted

class Solution {
    public int[] twoSum(int[] numbers, int target) {
        int[] res = new int[2];
        int l = 0;
        int r = numbers.length - 1;

        while (l < r) {

            if (numbers[l] + numbers[r] < target) {
                l++;
            } else if (numbers[l] + numbers[r] > target) {
                r--;
            } else {
                res[0] = l + 1;
                res[1] = r + 1;
                break;
            }

        }

        return res;

    }

    public static void main(String[] args) {
        int[] nums = {2, 7, 11, 15};
        int[] ints = new Solution().twoSum(nums, 9);
    }


}

11. Container With Most Water

class Solution {
    public int maxArea(int[] height) {
        int l = 0;
        int r = height.length - 1;
        int max = 0;
        while (l<r){
            int cur = 0;
            if (height[l] <= height[r]) {
                cur = height[l] * (r - l);
                max = Math.max(max,cur);
                l++;
            }else {
                cur = height[r] * (r - l);
                max = Math.max(max,cur);
                r--;
            }
        }
        return max;
    }
}

209. Minimum Size Subarray Sum

class Solution {
    public int minSubArrayLen(int target, int[] nums) {
        //nums[l...r]为滑动窗口
        int l = 0;
        int r = -1;
        int sum = 0;
        int res = nums.length + 1;
        while (l < nums.length) {
            if( r + 1 < nums.length &&sum < target) {
                r++;
                sum += nums[r];
            }else {
                sum -= nums[l];
                l++;
            }
            
            if (sum>=target){
                res = Math.min(res, r - l + 1);
            }

        }
        if( res == nums.length + 1){
            return 0;
        }
        return res;
                
        
    }
}

3. Longest Substring Without Repeating Characters

class Solution {
    public int lengthOfLongestSubstring(String s) {

        int[] freq = new int[256];
        //s[l...r]为滑动窗口
        int l = 0;
        int r = -1;
        int res = 0;
        while (l < s.length()){
            if ( r+1 <s.length() && freq[s.charAt(r+1)] == 0) {
                r++;
                freq[s.charAt(r)]++;
            } else {
                freq[s.charAt(l)]--;
                l++;
            }
            res = Math.max(res, r - l + 1);
        }
        return res;
    }
}

349. Intersection of Two Arrays

import java.util.HashSet;

class Solution {
    public int[] intersection(int[] nums1, int[] nums2) {
        HashSet<Integer> record = new HashSet<>();
        HashSet<Integer> resultSet = new HashSet<>();
        
        for (int i : nums1) {
            record.add(i);
        }
        for (int i : nums2) {
            if(record.contains(i)){
                resultSet.add(i);
            }
        }
        int i = 0;
        int[] res = new int[resultSet.size()];
        for (Integer re : resultSet) {
            res[i] = re;
            i++;
        }
        return res;
    }
}
